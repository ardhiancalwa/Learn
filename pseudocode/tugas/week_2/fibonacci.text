function fibonacci(n : int) -> int 
{mengembalikan fungsi nilai fibonacci}
kamus
    jumFibonacci, b1, b2, b2Lama, i : integer
algoritma
    jumFibonacci <- 0
    b1 <- 0
    b2 <- 0

    for i <- 1 to n do 
        jumFibonacci <- jumFibonacci + b1
        b2Lama <- b2
        b2 <- b1 + b2
        b1 <- b2Lama
    endfor
    return jumFibonacci
endfunction

program mencari_fibonacci
kamus
    n : integer
algoritma
    input(n)
    output(fibonacci(n))
endprogram
