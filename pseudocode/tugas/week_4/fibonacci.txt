function fibonacci(n : integer) -> integer
{
    mengembalikan nilai deret fibonacci hingga suku ke-10
}
algoritma
    if n == 0 or n == 1 then
        return n
    else
        return fibonacci(n - 1) + fibonacci(n - 2)
    endif
endfunction

kamus
    i : integer
algoritma
    for i <- 0 to 10 do
        output(fibonacci(i), " ")
    endfor
endprogram
